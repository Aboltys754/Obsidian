```
git # посмотреть список доступных команд

git --version  # проверьте версию Git

git help <Команда> # Получение справки о команде в web интерфейсе
или так 
git <Команда> --help

----------------------------------------------------------------------

git remote -v # Посмотреть какие удаленные репозитории подключены

git remote add <shortname> <url> Добовление удаленного репозитория
	<shortname> -переменная которую потом можно использовать как имя
	<url> ссылка на удаленный репозиторий

git fetch <remote> # Получить изменения из удаленной ветки но не сливать их

git remote show <remote> # Получить информацию о удаленном репозитории

git remote rename <old name> <new name> # Изменить имя ветки

git remote remove <name branch> # удаление ветки


-----------------------------------------------------------------------

git checkout -b <branch-name> # Создать ветку и перейти в нее

git branch -d <название ветки> # Удалить ветку

git branch <branch-name> # Создать ветку

git checkout <название ветки> # Перейти на ветку 


------------------------------------------------------------------------

# Слияние в новой ветке 
git checkout main # переход на новую ветку

git merge new-feature # влить изменения с new-feature в ветку main

git branch -d new-feature

git merge --no-ff <branch> # объединяет указанную ветвь с текущей ветвью, но всегда    генерирует коммит слияния


------------------------------------------------------------------------


 git config --global user.name "github_username"
 git config --global user.email "email_address"










```
